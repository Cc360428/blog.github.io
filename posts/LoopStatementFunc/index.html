package main

import (
	"fmt"
)

func main() {
	for i := 1; i <= 9; i++ {
		for j := 1; j <= i; j++ {
			fmt.Printf("%d*%d=%d  ", j, i, j*i)
		}
		fmt.Println()
	}
}

讲解：第二层来控制第二层循环次数，当第二层循环j=1时 判断 刚好i 也等于1 就循环一次，一次类推

package main

import (
	"github.com/Cc360428/HelpPackage/UtilsHelp/logs"
)

func main() {
	array := []int{1, 3, 4, 7, 2}
	logs.Info(array)
	BubbleASort(array)
	BubbleZSort(array)

}

func BubbleASort(values []int) {
	for i := 0; i < len(values)-1; i++ {
		for j := i + 1; j < len(values); j++ {
			if values[i] > values[j] {
				values[i], values[j] = values[j], values[i]
			}
		}
	}
	logs.Info(values)
}

func BubbleZSort(values []int) {
	for i := 0; i < len(values)-1; i++ {
		for j := i + 1; j < len(values); j++ {
			if values[i] < values[j] {
				values[i], values[j] = values[j], values[i]
			}
		}
	}
	logs.Info(values)
}

package main

import "github.com/Cc360428/HelpPackage/UtilsHelp/logs"

func main() {
	// 你放这里就成无限循环，j == 5 就执行下面代码LCC:
	for i := 0; i <= 10; i++ {
		for j := 0; j <= 10; j++ {
			logs.Info(i, j)
			if j == 5 {
				goto LCC
			}
		}
	}
LCC:
	logs.Info("LCC知识标签而已除了内置关键字，跳出循环")
}
